	id = 1000111

	desc = "We are pleased to inform that the ownership of Herzegovina has been passed to us."
	picture = "treaty"
	
	trigger = {
		owns = 783
		NOT = {
			tag = TUR
		}
		TUR = {
			any_owned_province = { region = TUR_3636 }
			NOT = {
				owns = 783 #Sarajevo
				owns = 804 #Novi Pazar
				# owns = 2177 #Pljevlja
				# owns = 2569 #Niksic
			}
		}
	}
			
	option = {
		name = "Great!"
		TUR = {
			any_owned = {
				limit = {
					region = TUR_3636
				}
				secede_province = THIS
			}
		}
	}
}
