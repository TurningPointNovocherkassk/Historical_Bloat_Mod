				factor = -2

# Province event after the formation of the Baha'i faith - missionaries in $PROVINCENAME$, results in religious insurrection

	id = 190303
	title = "EVTNAME190303"
	desc = "EVTDESC190303"
	
	trigger = {
        OR = {
			owner = { tag = PER }
            owner = { any_neighbor_country = { tag = PER } }
        }
        OR = {   
            owner = { is_culture_group = arab }
            owner = { is_culture_group = turko_semitic }
            owner = { is_culture_group = iranian }
            owner = { is_culture_group = turanian }
        }
        controlled_by_rebels = no
        NOT = { has_province_modifier = religious_insurrection }
        NOT = { year = 1892 }
        has_global_flag = bahaullah_has_appeared
    }
	
	mean_time_to_happen = {
		months = 1500
		modifier = {
			factor = 0.75
			owner = { has_country_flag = bahaullah_in_country }
		}
		modifier = {
			factor = 0.75
			owner = { tag = PER }
		}
		modifier = {
			factor = 0.95
			owner = { any_neighbor_country = { has_country_flag = bahaullah_in_country } }
		}
	}

		name = "Won't they just go away?"

		}
